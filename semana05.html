                import 'dart:io';

                bool esPrimo(int num) {
                  if (num < 2) return false;
                  for (int i = 2; i <= num ~/ 2; i++) {
                    if (num % i == 0) return false;
                  }
                  return true;
                }
                
                void main() {
                  print('Ingrese el primer número: ');
                  int inicio = int.parse(stdin.readLineSync()!);
                  print('Ingrese el segundo número: ');
                  int fin = int.parse(stdin.readLineSync()!);
                  
                  int sumaPrimos = 0;
                
                  for (int i = inicio; i <= fin; i++) {
                    if (esPrimo(i)) {
                      sumaPrimos += i;
                    }
                  }
                
                  print('La sumatoria de los números primos entre $inicio y $fin es: $sumaPrimos');
                }

                List> matriz1 = List.generate(n, (_) => List.filled(n, 0));

                List> suma = List.generate(n, (_) => List.filled(n, 0));

                print('Ingrese elementos de la primera matriz: ');
                for (int i = 0; i < n; i++) {
                    for (int j = 0; j < n; j++) {
                    matriz1[i][j] = int.parse(stdin.readLineSync()!);
                    }
                }

                print('Ingrese elementos de la segunda matriz: ');
                for (int i = 0; i < n; i++) {
                    for (int j = 0; j < n; j++) {
                    matriz2[i][j] = int.parse(stdin.readLineSync()!);
                    }
                }

                for (int i = 0; i < n; i++) {
                    for (int j = 0; j < n; j++) {
                    suma[i][j] = matriz1[i][j] + matriz2[i][j];
                    }
                }

                print('La matriz resultante de la suma es: ');
                for (var fila in suma) {
                    print(fila);
                    }
                }

                import 'dart:io';

                void main() {
                print('Ingrese el tamaño de la matriz: ');
                int n = int.parse(stdin.readLineSync()!);
                
                List> matriz = List.generate(n, (_) => List.filled(n, 0));
                
                int valor = 1;
                int inicioFila = 0, finFila = n - 1;
                int inicioCol = 0, finCol = n - 1;

                while (inicioFila <= finFila && inicioCol <= finCol) {
                    for (int i = inicioCol; i <= finCol; i++) {
                    matriz[inicioFila][i] = valor++;
                    }
                    inicioFila++;
                    
                    for (int i = inicioFila; i <= finFila; i++) {
                    matriz[i][finCol] = valor++;
                    }
                    finCol--;
                    
                    if (inicioFila <= finFila) {
                    for (int i = finCol; i >= inicioCol; i--) {
                        matriz[finFila][i] = valor++;
                    }
                    finFila--;
                    }
                    
                    if (inicioCol <= finCol) {
                    for (int i = finFila; i >= inicioFila; i--) {
                        matriz[i][inicioCol] = valor++;
                    }
                    inicioCol++;
                    }
                }

                print('Matriz en forma de espiral:');
                for (var fila in matriz) {
                    print(fila);
                    }
                }
